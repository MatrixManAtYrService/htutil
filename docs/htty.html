34def _check_installation_type_and_warn() -> None:
35    """
36    Check if this is a source installation without bundled ht binary and warn user.
37
38    This helps users who install from sdist on unsupported platforms understand
39    why htty might not work and what they need to do.
40    """
41    # Skip warning if user has explicitly set HTTY_HT_BIN
42    if os.environ.get("HTTY_HT_BIN"):
43        return
44
45    # Check if we have a bundled binary (wheel installation)
46    try:
47        bundled_files = impresources.files(_bundled)
48        ht_resource = bundled_files / "ht"
49
50        if ht_resource.is_file():
51            # We have a bundled binary, this is a wheel installation - no message needed
52            return
53
54    except FileNotFoundError:
55        arch = platform.machine()
56        system = platform.system()
57        msg = (
58            "htty requires a build of 'ht' from this fork: https://github.com/MatrixManAtYrService/ht "
59            "Wheel distributions of htty bundle it, but source distributions do not. "
60            f"You appear to have installed a source distribution, likely because no wheel is available "
61            f"for {arch} {system}.\n"
62            "If you'd like a wheel for your platform please open an issue: https://github.com/MatrixManAtYrService/htty/issues.\n"
63            "Available wheels are listed here: https://pypi.org/project/htty/#files\n\n"
64        )
65
66        # Check if ht is available in system PATH
67        ht_path = shutil.which("ht")
68        if ht_path:
69            msg += f"If {ht_path} is not from this fork, you may encounter issues. "
70        else:
71            msg += "`ht` is not in the PATH. To use htty, you'll need to build/install it separately. "
72
73        msg += "To override the `ht` binary used by htty, set the `HTTY_HT_BIN`. \n\n"
74
75        warnings.warn(
76            msg,
77            UserWarning,
78            stacklevel=2,
79        )

108@contextmanager
109def ht_binary() -> Iterator[HTBinary]:
110    """
111    Context manager that provides access to the ht binary.
112
113    Order of precedence:
114    1. HTTY_HT_BIN environment variable (if set and valid) - user override
115    2. Bundled ht binary via importlib.resources (production/wheel)
116    3. System 'ht' command from PATH (development fallback)
117
118    Usage:
119        with ht_binary() as ht:
120            cmd = ht.build_command("--help")
121            ht_proc = ht.run_subprocess("--version")
122
123    Raises:
124        RuntimeError: If ht binary cannot be found anywhere
125    """
126
127    # Try each strategy in order
128    for strategy in [_try_user_specified_binary, _try_bundled_binary, _try_system_binary]:
129        try:
130            ht_path = strategy()
131            if ht_path:
132                yield HTBinary(ht_path)
133                return
134        except RuntimeError:
135            # Re-raise user configuration errors immediately
136            raise
137
138    # If we get here, ht is not available anywhere
139    import platform
140
141    arch = platform.machine()
142    system = platform.system()
143
144    raise RuntimeError(
145        f"Could not find ht binary for {system} {arch}.\n\n"
146        "This likely happened because:\n"
147        " You installed from source (no bundled ht binary)\n"
148        " No pre-built wheel available for your architecture\n\n"
149        "Solutions:\n"
150        "1. Install ht separately: https://github.com/andyk/ht\n"
151        "   Then ensure it's in your PATH\n"
152        "2. Set HTTY_HT_BIN to point to ht binary:\n"
153        "   export HTTY_HT_BIN=/path/to/ht\n"
154        "3. Use a supported architecture with pre-built wheels:\n"
155        "    Linux: x86_64, aarch64\n"
156        "    macOS: x86_64, arm64\n\n"
157        "For help: https://github.com/MatrixManAtYrService/htty/issues"
158    )

def run( command: Union[str, List[str]], rows: Optional[int] = None, cols: Optional[int] = None, no_exit: bool = True) -> htty.ht.HTProcess: 
605def run(
606    command: Union[str, List[str]],
607    rows: Optional[int] = None,
608    cols: Optional[int] = None,
609    no_exit: bool = True,
610) -> HTProcess:
611    """
612    Run a command using the 'ht' tool and return a HTProcess object
613    that can be used to interact with it.
614
615    Args:
616        command: The command to run (string or list of strings)
617        rows: Number of rows for the terminal size (height)
618        cols: Number of columns for the terminal size (width)
619        no_exit: If True, use the --no-exit flag to keep ht running after subprocess exits
620
621    Returns:
622        An HTProcess instance
623    """
624
625    # Handle both string commands and pre-split argument lists
626    if isinstance(command, str):
627        cmd_args = command.split()
628    else:
629        cmd_args = command
630
631    # Create a queue for events
632    event_queue: Queue[Dict[str, Any]] = queue.Queue()
633
634    # Use the ht binary context manager to start the process
635    with ht_binary() as ht:
636        # Build the ht command with event subscription
637        ht_cmd_args = [
638            "--subscribe",
639            "init,snapshot,output,resize,pid,exitCode",
640        ]
641
642        # Add size options if specified
643        if rows is not None and cols is not None:
644            ht_cmd_args.extend(["--size", f"{cols}x{rows}"])
645
646        # Add no-exit option if specified
647        if no_exit:
648            ht_cmd_args.append("--no-exit")
649
650        # Add separator and the command to run
651        ht_cmd_args.append("--")
652        ht_cmd_args.extend(cmd_args)
653
654        # Log the exact command for debugging
655        full_cmd = ht.build_command(*ht_cmd_args)
656        logger = logging.getLogger(__name__)
657        logger.debug(f"Executing ht command: {' '.join(full_cmd)}")
658
659        # Launch ht using the HTBinary helper
660        ht_proc = ht.run_subprocess(
661            *ht_cmd_args,
662            stdin=subprocess.PIPE,
663            stdout=subprocess.PIPE,
664            stderr=subprocess.PIPE,
665            text=True,
666            bufsize=1,
667        )
668
669    # Create a reader thread to capture ht output
670    def reader_thread(
671        ht_proc: subprocess.Popen[str],
672        queue_obj: Queue[Dict[str, Any]],
673        ht_process: HTProcess,
674    ) -> None:
675        while True:
676            if ht_proc.stdout is None:
677                break
678            line = ht_proc.stdout.readline()
679            if not line:
680                break
681            line = line.strip()
682            if not line:
683                continue
684
685            try:
686                event = json.loads(line)
687                queue_obj.put(event)
688
689                if event["type"] == "output":
690                    ht_process.output_events.append(event)
691                elif event["type"] == "exitCode":
692                    ht_process.subprocess_exited = True
693                    if hasattr(ht_process, "subprocess_controller"):
694                        exit_code = event.get("data", {}).get("exitCode")
695                        if exit_code is not None:
696                            ht_process.subprocess_controller.exit_code = exit_code
697                        else:
698                            logger = logging.getLogger(__name__)
699                            logger.error(f"Received exitCode event without valid exit code data: {event}")
700            except json.JSONDecodeError as e:
701                logger = logging.getLogger(__name__)
702                logger.error(f"Received non-JSON output from ht process: {line!r}")
703                raise RuntimeError(f"ht process produced unexpected non-JSON output: {line!r}") from e
704
705    # Create an HTProcess instance
706    process = HTProcess(
707        ht_proc,
708        event_queue,
709        command=" ".join(cmd_args),
710        rows=rows,
711        cols=cols,
712        no_exit=no_exit,
713    )
714
715    # Start the reader thread
716    thread = threading.Thread(target=reader_thread, args=(ht_proc, event_queue, process), daemon=True)
717    thread.start()
718    # Wait briefly for the process to initialize
719    start_time = time.time()
720    while time.time() - start_time < 2:
721        try:
722            event = event_queue.get(block=True, timeout=0.5)
723            if event["type"] == "pid":
724                pid = event["data"]["pid"]
725                process.subprocess_controller.pid = pid
726                break
727        except queue.Empty:
728            continue
729
730    sleep(DEFAULT_SLEEP_AFTER_KEYS)
731    return process
